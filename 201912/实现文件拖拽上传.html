点击或拖拽打开本地文件浏览

   <div id="container">
        <input type="file" id="files" multiple>
        <p>+</p>

技巧是 将<input> 撑满父容器并设置全透明，并用z-index浮在最上层，拖拽事件如下

const target = document.querySelector("#container");
target.addEventListener("dragover", handleDropover);
target.addEventListener("drop", handleDrop);

function handleDropover(event) {
  event.stopPropagation(); //阻止冒泡传递和默认行为
  event.preventDefault();
}

function handleDrop(event) {
  event.stopPropagation();
  event.preventDefault();
  /***** 访问拖拽文件 *****/
  const files = event.dataTransfer.files;
  readFile(files)
}

document.querySelector('#files').addEventListener('change',handleChang,false)
function handleChang(event){
    var files = event.target.files;
    readFile(files)
}

function readFile(files){
    if(files.length==0){
        return
    }
    for (const file of files) {
        const reader = new FileReader();
       
        reader.onload = function(event){
            console.log(event.target.result)
            //此处略
        }
        
        reader.readAsDataURL(file); // 读取文件为DataURL,适合图片预览直接赋给img标签
        //reader.readAsArrayBuffer(file) //二进制对象
        //reader.readAsBinaryString(file) //二进制字符串，上传到服务器
        //reader.readAsText(file)  //用于文本文件
    }
}
