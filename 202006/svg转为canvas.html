# svg转为canvas

先上案例代码：

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>svg转为canvas</title>
    <style>
        .box{
            border:1px solid red;
            width: 300px;
            height: 300px;
            float: left;
            margin: 50px;
        }
    </style>
</head>
<body>
    <div class="box">
        <svg width="100%" height="100%" id="svg">
           
              <rect x="0" y="0" width="50" height="50" fill="#feac5e">
                   
             </rect>
        </svg>
    </div>

<div class="box">
    <canvas id="canvas" width="100%" height="100%">

    </canvas>
</div>

</body>
</html>

<script>
    
    var svgStr = new XMLSerializer().serializeToString(document.querySelector('#svg'))

    var img = new Image();

    var svg = new Blob([svgStr], {type: "image/svg+xml;charset=utf-8"});
    //console.log(svg)  

  
    img.src = self.URL.createObjectURL(svg)  //self == window

    img.onload = function(){
        var canvas = document.querySelector('#canvas')
        var ctx = canvas.getContext('2d')
        ctx.drawImage(img,0,0)
    }

</script>

说明：

步骤1：svg 本身是一段XMl标记，通过XMLSerializer().serializeToString() 获取源码

步骤2：新建一个Image对象，将svg指定到它的src属性，之前需要通过Blob和URL映射一个临时的虚拟url

步骤3：等待加载完成onload,在画布canvas绘制Image
