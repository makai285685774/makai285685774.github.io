# 实现js 模式的MessageBox的弹窗vue组件

组件内 messagebox( opts ) 直接调用，而不是通多v-if实现组件的显示或隐藏

1. 模板视图

//index.vue文件
<template>
    <div class="message_box">
        <h2>{{ title }}</h2>
        <div class="content">
            内容
        </div>
        <div class="footer">
            <div @click="handleOk">{{ ok }}</div>
            <div @click="handleCancel">{{ cancel }}</div>
        </div>
    </div>
</template>
<script>
export default {
    name:'MessageBox'
}
</script>
<style scoped>
    ...  //这里添加样式，略
</style>

无需赘述，上面的代码是messagebox弹窗的基本结构，上中下三部分

  1. 实例化组件**
//index.js文件

import MessageBox from './index.vue' //引入模板
import Vue from 'vue'

export var messageBox = (function () { // 采用闭包的形式对外暴露函数名

    let defaults = { //默认配置
        title: '标题',
        ok:'ok',
        cancel:'cancel',
        handleOk:null,
        handleCancel:null
    }

    var MyComponent = Vue.extend(MessageBox)  //构造器，相当于定义一个基础样式的类
  
    return function (opts) {  //闭包

        for (var key in opts) {  //覆盖默认配置
            defaults[key] = opts[key]
        }
        
        var vm = new MyComponent({  //实例化
            el:document.createElement('div'),  //为实例挂载到dom
            data(){
                return {
                    title:defaults.title,
                    ok:defaults.ok,
                    cancel:defaults.cancel
                }
            },
            methods: {
                handleOk(){
                    defaults.handleOk && defaults.handleOk.call(this)
                    document.body.removeChild(vm.$el) //移除容器
                },
                handleCancel(){
                    defaults.handleCancel && defaults.handleCancel.call(this)
                    document.body.removeChild(vm.$el) //移除容器
                }
            }

        })

        document.body.appendChild(vm.$el)    //将实例元素插入dom结构
    }

})()
